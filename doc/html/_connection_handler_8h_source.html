135  template <typename Request, typename Response>
136  Status
137  apiHandler(const std::function<Status(void)>& validate, std::function<Status(Request&)> pre,
138  Status (MilvusConnection::*rpc)(const Request&, Response&, const GrpcOpts&),
139  std::function<Status(const Response&)> wait_for_status, std::function<Status(const Response&)> post) {
140  if (connection_ == nullptr) {
141  return {StatusCode::NOT_CONNECTED, "Connection is not created!"};
142  }
143 
144  // validate input
145  if (validate) {
146  auto status = validate();
147  if (!status.IsOk()) {
148  return status;
149  }
150  }
151 
152  // construct rpc request
153  Request rpc_request;
154  if (pre) {
155  auto status = pre(rpc_request);
156  if (!status.IsOk()) {
157  return status;
158  }
159  }
160 
161  // call rpc interface
162  Response rpc_response;
163  // the timeout value can be changed by MilvusClient::SetRpcDeadlineMs()
164  uint64_t timeout = connection_->GetConnectParam().RpcDeadlineMs();
165  auto func = std::bind(rpc, connection_.get(), rpc_request, std::placeholders::_1, GrpcOpts{timeout});
166  auto caller = [&func, &rpc_response]() { return func(rpc_response); };
167  auto status = Retry(caller, retry_param_);
168  if (!status.IsOk()) {
169  // response's status already checked in connection class
170  return status;
171  }
172 
173  // wait loop
174  if (wait_for_status) {
175  status = wait_for_status(rpc_response);
176  if (!status.IsOk()) {
177  return status;
178  }
179  }
180 
181  // process results
182  if (post) {
183  status = post(rpc_response);
184  if (!status.IsOk()) {
185  return status;
186  }
187  }
188  return Status::OK();
189  }
190 
191  private:
192  MilvusConnectionPtr connection_;
193  RetryParam retry_param_;
194 };
